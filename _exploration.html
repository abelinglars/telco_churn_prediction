Code 
def snake_to_title(snake):
  words = [word.title() for word in snake.split("_")]
  title = " ".join(words)
  return title
  
for col in df.select_dtypes(include = "number"):

  title = snake_to_title(col)

  sns.displot(df[col]).set(title = f"Distribution of {title}")
